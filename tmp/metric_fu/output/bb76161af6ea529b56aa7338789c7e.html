253  def pair_program_expense
  
254    program = Program.new(id: 1, name: 'program test',
  
255                          description: 'Program to make a test')
  
256    expense = Expense.new(document_number: '0000',
  
257                          payment_date: Date.new(2015, 01, 01), value: 100,
  
258                          program_id: 1)
  
259    [program, expense]
  
260  end

262  def generate_program_seed
  
263    SuperiorPublicAgency.create(id: 1, name: 'SuperiorPublicAgency')
  
264    PublicAgency.create(id: 1, name: 'PublicAgency1', views_amount: 10,
  
265                        superior_public_agency_id: 1)
  
266    name_program = %w(Programa1 Programa2)
  
267    j = 0
  
268    for i in 1..2
  
269      date = Date.new(2015, i, i)
  
270      program = Program.create(id: i, name: name_program[i - 1],
  
271                               description: 'Outros')
  
272      2.times do
  
273        Expense.create(id: j, document_number: i, payment_date: date,
  
274                       value: i + 5, program_id: i, public_agency_id: 1)
  
275        j += 1
  
276      end
  
277    end
  
278    Expense.create(id: 10, document_number: 100,
  
279                   payment_date: Date.new(2015, 1, 1), value: 0, program_id: 1,
  
280                   public_agency_id: 1)
  
281  end
