17  def self.add_budget_array(budget_years, data_budget)
  
18    budgets = data_budget['results']['bindings']
  
19    budgets.each do |budget|

25  def self.create_budget_year(budget)
  
26    year = budget['ano']['value']
  
27    value_budget = budget['somaProjetoLei']['value']
  
28    budget_by_year = {}

30      budget_by_year = { 'year' => year.to_i, 'value' => value_budget.to_i }

33  end

88  def self.generate_query(public_agency_id, year = 'Todos')
  
89    prefix = 'PREFIX rdf: <http://www.w3.org/1999/02/22-rdf-syntax-ns#> '\
  
90            'PREFIX loa: <http://vocab.e.gov.br/2013/09/loa#>'
  
91
  
92    year_query = query_for_year(year)
  
93
  
94    # puts year_query
  
95    query = 'SELECT ?ano, (SUM(?valorProjetoLei) AS ?somaProjetoLei) WHERE {'\
  
96            '?itemBlankNode loa:temExercicio ?exercicioURI . ' + year_query +
  
97            '?exercicioURI loa:identificador ?ano . '\
  
98            '?itemBlankNode loa:temUnidadeOrcamentaria ?uoURI . '\
  
 99            '?uoURI loa:codigo "' + public_agency_id + '" . '\
  
100            '?itemBlankNode loa:valorProjetoLei ?valorProjetoLei . }'
  
101
  
102    # puts query
  
103    url_query = URI.encode(prefix + query)
  
104
  
105    url_query
  
106  end

113    end
